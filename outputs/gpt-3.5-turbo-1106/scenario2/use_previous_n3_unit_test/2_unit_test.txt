```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

import "remix_tests.sol";
import "./YourContract.sol";

contract MedicarePolicyTest {
    // Test for coverage of IVIG if patient has a diagnosis of primary immune deficiency disease
    function testIVIGCoverage() public {
        YourContract contractInstance = new YourContract();
        bool hasCoverage = contractInstance.hasIVIGCoverage("primary immune deficiency disease");
        Assert.equal(hasCoverage, true, "IVIG coverage should be available for primary immune deficiency disease");
    }

    // Test for doctor's determination of medical appropriateness for patient to receive IVIG in their home
    function testMedicalAppropriateness() public {
        YourContract contractInstance = new YourContract();
        bool isMedicallyAppropriate = contractInstance.isIVIGMedicallyAppropriate("patientAddress");
        Assert.equal(isMedicallyAppropriate, true, "IVIG should be medically appropriate for the patient to receive in their home");
    }

    // Test for Medicare Part B coverage of IVIG cost
    function testPartBCoverage() public {
        YourContract contractInstance = new YourContract();
        bool hasPartBCoverage = contractInstance.hasPartBCoverage("IVIG");
        Assert.equal(hasPartBCoverage, true, "Medicare Part B should cover the cost of IVIG");
    }

    // Test for non-coverage of other items and services related to patient receiving IVIG at home
    function testNonCoverage() public {
        YourContract contractInstance = new YourContract();
        bool hasNonCoverage = contractInstance.hasNonCoverage("IVIG");
        Assert.equal(hasNonCoverage, true, "Medicare Part B should not cover other items and services related to patient receiving IVIG at home");
    }
}
```
