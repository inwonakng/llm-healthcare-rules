# Rule 1: Requirements for Medicare coverage of transplant drug therapy

medicare:helpedPayForTransplant rdf:type rdf:Property ;
    rdfs:label "Medicare must have helped pay for your organ transplant." .

medicare:hasPartACoverage rdf:type rdf:Property ;
    rdfs:label "You must have Part A coverage at the time of the covered transplant." .

medicare:hasPartBCoverage rdf:type rdf:Property ;
    rdfs:label "You must have Part B coverage at the time you receive immunosuppressive drugs." .

medicare:hasPartBCoverageImmunosuppressive rdf:type rdf:Property ;
    rdfs:label "If Part B doesn't cover immunosuppressive drugs, Medicare Part D (drug coverage) can cover them." .

medicare:hasOriginalMedicare rdf:type rdf:Property ;
    rdfs:label "If you have Original Medicare, you can join a Medicare drug plan for coverage." .

medicare:coverageEndsESRD rdf:type rdf:Property ;
    rdfs:label "For individuals with Medicare due to End-Stage Renal Disease (ESRD), coverage ends 36 months after a successful kidney transplant." .

medicare:losesPartAAfter36Months rdf:type rdf:Property ;
    rdfs:label "If you lose Part A coverage 36 months after a kidney transplant and don't have certain types of other health coverage, Medicare offers a benefit that covers immunosuppressive drugs only." .

medicare:benefitNotSubstitute rdf:type rdf:Property ;
    rdfs:label "This benefit is not a substitute for full health coverage." .

medicare:canSignUpAfterPartAEnds rdf:type rdf:Property ;
    rdfs:label "You can sign up for this benefit after your Part A coverage ends by contacting Social Security." .

medicare:callTTYUsers rdf:type rdf:Property ;
    rdfs:label "TTY users can call 1-800-325-0788 for assistance." .

# Rule 2: Costs for the immunosuppressive drug benefit in 2023

medicare:monthlyPremium rdf:type rdf:Property ;
    rdfs:label "A monthly premium of $97.10*." .

medicare:deductible rdf:type rdf:Property ;
    rdfs:label "$226 deductible." .

medicare:payPercentage rdf:type rdf:Property ;
    rdfs:label "After meeting the deductible, you'll pay 20% of the Medicare-approved amount for immunosuppressive drugs." .

# Rule 3: Final decision based on logic rules

medicare:qualifyForBenefit rdf:type rdf:Property ;
    rdfs:label "To qualify for the benefit, the following requirements for Medicare coverage of transplant drug therapy should be met." ;
    rdfs:subPropertyOf medicare:helpedPayForTransplant,
                        medicare:hasPartACoverage,
                        medicare:hasPartBCoverage,
                        medicare:hasPartBCoverageImmunosuppressive,
                        medicare:hasOriginalMedicare,
                        medicare:coverageEndsESRD,
                        medicare:losesPartAAfter36Months,
                        medicare:benefitNotSubstitute,
                        medicare:canSignUpAfterPartAEnds,
                        medicare:callTTYUsers .
