@prefix rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#> .
@prefix rdfs: <http://www.w3.org/2000/01/rdf-schema#> .
@prefix policy: <http://example.com/policy#> .
@prefix condition: <http://example.com/condition#> .
@prefix decision: <http://example.com/decision#> .

# Define the policy document
policy: a rdf:Document ;
    rdfs:comment "Intravenous Immune Globulin (IVIG) provided in home: Medicare covers IVIG if you have a diagnosis of primary immune deficiency disease, and your doctor decides that it’s medically appropriate for you to get the IVIG in your home. Part B covers the IVIG itself, but doesn’t pay for other items and services related to you getting the IVIG at home." .

# Define the conditions
condition:PrimaryImmuneDeficiencyDisease a rdf:Property ;
    rdfs:comment "The patient has a diagnosis of primary immune deficiency disease." .

condition:MedicallyAppropriate a rdf:Property ;
    rdfs:comment "The doctor determines that it is medically appropriate for the patient to receive IVIG in their home." .

# Define the decisions
decision:CoverIVIG a rdf:Property ;
    rdfs:comment "Medicare covers the cost of Intravenous Immune Globulin (IVIG)." .

decision:CoverRelatedItems a rdf:Property ;
    rdfs:comment "Medicare does not cover other items and services related to receiving IVIG at home." .

# Define the logic rules
{
    ?patient condition:PrimaryImmuneDeficiencyDisease true ;
             condition:MedicallyAppropriate true .
} => {
    ?patient decision:CoverIVIG true ;
             decision:CoverRelatedItems false .
} .
