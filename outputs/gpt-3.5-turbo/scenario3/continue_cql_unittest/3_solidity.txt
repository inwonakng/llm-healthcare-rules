pragma solidity ^0.8.0;

contract InsurancePolicy {
    struct Condition {
        string code;
        string display;
        string payor;
        string subjectCoverage;
        string onsetString;
    }

    struct MedicationRequest {
        string medicationCode;
        string medicationDisplay;
        string subjectCoverage;
    }

        string coverage;
    }

        string id;

    // Named expressions based on the insurance policy document

    // Requirement 1: Medicare covers transplant drug therapy if Medicare helped pay for your organ transplant.
    function medicareCoversTransplantDrugTherapy() public pure returns (bool) {
        Condition memory condition;
        condition.code = "organ transplant";
        condition.display = "Organ Transplant";
        condition.payor = "Medicare";

        return (keccak256(abi.encodePacked(condition.code)) == keccak256(abi.encodePacked("organ transplant")) &&
            keccak256(abi.encodePacked(condition.payor)) == keccak256(abi.encodePacked("Medicare")));
    }

        Condition memory condition;

        condition.display = "Organ Transplant";

        return (keccak256(abi.encodePacked(condition.code)) == keccak256(abi.encodePacked("organ transplant")) &&
            keccak256(abi.encodePacked(condition.subjectCoverage)) == keccak256(abi.encodePacked("Part A")));
    }
